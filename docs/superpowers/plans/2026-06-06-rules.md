# Rules Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a rules system so teams, agents, and LLMs can create and maintain prescriptive rules (prompt wording, processing steps, domain conventions) stored at team/category/user scope, and automatically inject applicable rules into any prompt via a `prompt_enhance` MCP tool.

**Architecture:** Rules are stored in a new SQLite `rules` table. A new `RuleStore` interface on `*SQLiteStore` handles CRUD and scope-filtered lookup. Five CRUD MCP tools (`rule_store`, `rule_get`, `rule_list`, `rule_update`, `rule_delete`) and one prompt tool (`prompt_enhance`) are registered. `prompt_enhance` fetches all enabled rules matching the caller's team/category/user context, orders them team → category → user (broad to specific), and prepends them to the original prompt.

**Tech Stack:** Go, SQLite (existing WAL-mode database), mark3labs/mcp-go v0.54.1 (existing), github.com/google/uuid (existing).

### Background

`RuleStore` is a separate interface (not extending `Store` or `AnalysisStore`). `*SQLiteStore` implements all three interfaces; a compile-time check enforces this. The `rules` table stores a boolean `enabled` as `INTEGER` (0/1) because SQLite has no native bool type. `GetApplicableRules` builds a dynamic WHERE clause so empty inputs (no team, no category, no user) never accidentally match rules that happen to have an empty `scope_value`.

```go
package storage

import (
	"context"
	"time"
)

type RuleScope string

const (
	RuleScopeTeam     RuleScope = "team"
	RuleScopeCategory RuleScope = "category"
	RuleScopeUser     RuleScope = "user"
)

// Rule is a prescriptive directive that governs how prompts are worded,
// what steps to follow, or how to process a request.
type Rule struct {
	ID         string
	Title      string
	Content    string
	Scope      RuleScope // team | category | user
	ScopeValue string    // team ID, domain/category name, or user/author identifier
	Priority   int       // higher = applied first within the same scope
	Enabled    bool
	Author     string
	CreatedAt  time.Time
	UpdatedAt  time.Time
}

// RuleFilter restricts which rules ListRules returns.
type RuleFilter struct {
	Scope      RuleScope
	ScopeValue string
	// Limit is max rules to return. 0 = default (50). Negative = unlimited.
	Limit int
}

// RuleStore is the persistence interface for rules.
type RuleStore interface {
	// StoreRule creates a new rule and returns its assigned UUID.
	StoreRule(ctx context.Context, rule Rule) (string, error)
	// GetRule returns a rule by ID, or ErrNotFound if absent.
	GetRule(ctx context.Context, id string) (*Rule, error)
	// ListRules returns rules matching the filter, ordered by priority desc.
	ListRules(ctx context.Context, filter RuleFilter) ([]Rule, error)
	// UpdateRule replaces all editable fields of an existing rule.
	// Returns ErrNotFound if the rule ID does not exist.
	UpdateRule(ctx context.Context, rule Rule) error
	// DeleteRule permanently removes a rule.
	// Returns ErrNotFound if the rule ID does not exist.
	DeleteRule(ctx context.Context, id string) error
	// GetApplicableRules returns all enabled rules matching the given team,
	// category, and user, ordered team → category → user then priority desc.
	// Empty strings are ignored (no rules fetched for that scope).
	GetApplicableRules(ctx context.Context, team, category, user string) ([]Rule, error)
}
```

```go
package storage

import (
	"context"
	"database/sql"
	"errors"
	"fmt"
	"strings"

	"github.com/google/uuid"
)

func (s *SQLiteStore) StoreRule(ctx context.Context, rule Rule) (string, error) {
	rule.ID = uuid.NewString()
	enabled := 1
	if !rule.Enabled {
		enabled = 0
	}
	_, err := s.db.ExecContext(ctx, `
		INSERT INTO rules (id, title, content, scope, scope_value, priority, enabled, author)
		VALUES (?, ?, ?, ?, ?, ?, ?, ?)
	`, rule.ID, rule.Title, rule.Content, string(rule.Scope), rule.ScopeValue,
		rule.Priority, enabled, rule.Author)
	if err != nil {
		return "", fmt.Errorf("insert rule: %w", err)
	}
	return rule.ID, nil
}

func (s *SQLiteStore) GetRule(ctx context.Context, id string) (*Rule, error) {
	row := s.db.QueryRowContext(ctx, `
		SELECT id, title, content, scope, scope_value, priority, enabled, author, created_at, updated_at
		FROM rules WHERE id = ?
	`, id)
	r, err := scanRuleRow(row.Scan)
	if err != nil {
		if errors.Is(err, sql.ErrNoRows) {
			return nil, ErrNotFound
		}
		return nil, fmt.Errorf("get rule: %w", err)
	}
	return r, nil
}

func (s *SQLiteStore) ListRules(ctx context.Context, filter RuleFilter) ([]Rule, error) {
	limit := filter.Limit
	if limit == 0 {
		limit = 50
	}

	query := `SELECT id, title, content, scope, scope_value, priority, enabled, author, created_at, updated_at
	          FROM rules WHERE 1=1`
	args := []any{}

	if filter.Scope != "" {
		query += " AND scope = ?"
		args = append(args, string(filter.Scope))
	}
	if filter.ScopeValue != "" {
		query += " AND scope_value = ?"
		args = append(args, filter.ScopeValue)
	}
	query += " ORDER BY priority DESC, created_at ASC"
	if limit > 0 {
		query += " LIMIT ?"
		args = append(args, limit)
	}

	rows, err := s.db.QueryContext(ctx, query, args...)
	if err != nil {
		return nil, fmt.Errorf("list rules: %w", err)
	}
	defer rows.Close()

	var rules []Rule
	for rows.Next() {
		r, err := scanRuleRow(rows.Scan)
		if err != nil {
			return nil, fmt.Errorf("scan rule: %w", err)
		}
		rules = append(rules, *r)
	}
	return rules, rows.Err()
}

func (s *SQLiteStore) UpdateRule(ctx context.Context, rule Rule) error {
	enabled := 1
	if !rule.Enabled {
		enabled = 0
	}
	res, err := s.db.ExecContext(ctx, `
		UPDATE rules
		SET title = ?, content = ?, scope = ?, scope_value = ?,
		    priority = ?, enabled = ?, author = ?,
		    updated_at = CURRENT_TIMESTAMP
		WHERE id = ?
	`, rule.Title, rule.Content, string(rule.Scope), rule.ScopeValue,
		rule.Priority, enabled, rule.Author, rule.ID)
	if err != nil {
		return fmt.Errorf("update rule: %w", err)
	}
	n, _ := res.RowsAffected()
	if n == 0 {
		return fmt.Errorf("rule %q: %w", rule.ID, ErrNotFound)
	}
	return nil
}

func (s *SQLiteStore) DeleteRule(ctx context.Context, id string) error {
	res, err := s.db.ExecContext(ctx, "DELETE FROM rules WHERE id = ?", id)
	if err != nil {
		return fmt.Errorf("delete rule: %w", err)
	}
	n, _ := res.RowsAffected()
	if n == 0 {
		return fmt.Errorf("rule %q: %w", id, ErrNotFound)
	}
	return nil
}

func (s *SQLiteStore) GetApplicableRules(ctx context.Context, team, category, user string) ([]Rule, error) {
	var conds []string
	var args []any

	if team != "" {
		conds = append(conds, "(scope = 'team' AND scope_value = ?)")
		args = append(args, team)
	}
	if category != "" {
		conds = append(conds, "(scope = 'category' AND scope_value = ?)")
		args = append(args, category)
	}
	if user != "" {
		conds = append(conds, "(scope = 'user' AND scope_value = ?)")
		args = append(args, user)
	}

	if len(conds) == 0 {
		return []Rule{}, nil
	}

	query := fmt.Sprintf(`
		SELECT id, title, content, scope, scope_value, priority, enabled, author, created_at, updated_at
		FROM rules
		WHERE enabled = 1 AND (%s)
		ORDER BY
		  CASE scope WHEN 'team' THEN 1 WHEN 'category' THEN 2 WHEN 'user' THEN 3 END ASC,
		  priority DESC,
		  created_at ASC
	`, strings.Join(conds, " OR "))

	rows, err := s.db.QueryContext(ctx, query, args...)
	if err != nil {
		return nil, fmt.Errorf("get applicable rules: %w", err)
	}
	defer rows.Close()

	var rules []Rule
	for rows.Next() {
		r, err := scanRuleRow(rows.Scan)
		if err != nil {
			return nil, fmt.Errorf("scan applicable rule: %w", err)
		}
		rules = append(rules, *r)
	}
	return rules, rows.Err()
}

// scanRuleRow scans a rule from a variadic Scan function (works for both *sql.Row and *sql.Rows).
func scanRuleRow(scan func(...any) error) (*Rule, error) {
	var r Rule
	var scopeStr string
	var enabledInt int
	var createdAt, updatedAt string
	err := scan(&r.ID, &r.Title, &r.Content, &scopeStr, &r.ScopeValue,
		&r.Priority, &enabledInt, &r.Author, &createdAt, &updatedAt)
	if err != nil {
		return nil, err
	}
	r.Scope = RuleScope(scopeStr)
	r.Enabled = enabledInt != 0
	r.CreatedAt = parseTimestamp(createdAt)
	r.UpdatedAt = parseTimestamp(updatedAt)
	return &r, nil
}

// Ensure *SQLiteStore implements RuleStore at compile time.
var _ RuleStore = (*SQLiteStore)(nil)
```

### Background

All 6 handlers follow the same pattern as `tools.go`: return a closure, use `req.GetString`/`req.GetInt`, return `mcplib.NewToolResultError` (not a Go error) on failures.

`prompt_enhance` returns the original prompt unchanged when no rules match — callers can always use the enhanced version without checking whether rules were applied.

`rule_update` requires the caller to supply all fields (id + title + content are mandatory; scope, scope_value, priority, enabled, author are optional with reasonable defaults). The description tells callers to use `rule_get` first.

`HandleRuleUpdate` uses a `getBoolArg` helper that accepts bool, float64, or string values from the MCP argument map (clients may serialize booleans differently).

```go
	getErr   error
	listErr  error
	updateErr error
	deleteErr error

func (m *mockRuleStore) StoreRule(_ context.Context, rule storage.Rule) (string, error) {
	if m.storeErr != nil {
		return "", m.storeErr
	}
	rule.ID = "rule-" + rule.Title
	m.rules = append(m.rules, rule)
	return rule.ID, nil
}

func (m *mockRuleStore) GetRule(_ context.Context, id string) (*storage.Rule, error) {
	if m.getErr != nil {
		return nil, m.getErr
	}
	for _, r := range m.rules {
		if r.ID == id {
			return &r, nil
		}
	}
	return nil, storage.ErrNotFound
}

func (m *mockRuleStore) ListRules(_ context.Context, _ storage.RuleFilter) ([]storage.Rule, error) {
	if m.listErr != nil {
		return nil, m.listErr
	}
	return m.rules, nil
}

func (m *mockRuleStore) UpdateRule(_ context.Context, rule storage.Rule) error {
	if m.updateErr != nil {
		return m.updateErr
	}
	for i, r := range m.rules {
		if r.ID == rule.ID {
			m.rules[i] = rule
			return nil
		}
	}
	return storage.ErrNotFound
}

func (m *mockRuleStore) DeleteRule(_ context.Context, id string) error {
	if m.deleteErr != nil {
		return m.deleteErr
	}
	for i, r := range m.rules {
		if r.ID == id {
			m.rules = append(m.rules[:i], m.rules[i+1:]...)
			return nil
		}
	}
	return storage.ErrNotFound
}

func (m *mockRuleStore) GetApplicableRules(_ context.Context, _, _, _ string) ([]storage.Rule, error) {
	var out []storage.Rule
	for _, r := range m.rules {
		if r.Enabled {
			out = append(out, r)
		}
	}
	return out, nil
}
```

```go
import (
	"context"

func HandleRuleStore(store storage.RuleStore) func(context.Context, mcplib.CallToolRequest) (*mcplib.CallToolResult, error) {
	return func(ctx context.Context, req mcplib.CallToolRequest) (*mcplib.CallToolResult, error) {
		title := req.GetString("title", "")
		content := req.GetString("content", "")
		scope := req.GetString("scope", "team")

		if title == "" || content == "" {
			return mcplib.NewToolResultError("title and content are required"), nil
		}
		if scope != "team" && scope != "category" && scope != "user" {
			return mcplib.NewToolResultError("scope must be one of: team, category, user"), nil
		}

		rule := storage.Rule{
			Title:      title,
			Content:    content,
			Scope:      storage.RuleScope(scope),
			ScopeValue: req.GetString("scope_value", ""),
			Priority:   req.GetInt("priority", 0),
			Enabled:    true,
			Author:     req.GetString("author", ""),
		}

		id, err := store.StoreRule(ctx, rule)
		if err != nil {
			return mcplib.NewToolResultError(fmt.Sprintf("store failed: %v", err)), nil
		}
		return mcplib.NewToolResultText(fmt.Sprintf("stored rule with id=%s", id)), nil
	}
}

func HandlePromptEnhance(store storage.RuleStore) func(context.Context, mcplib.CallToolRequest) (*mcplib.CallToolResult, error) {
	return func(ctx context.Context, req mcplib.CallToolRequest) (*mcplib.CallToolResult, error) {
		prompt := req.GetString("prompt", "")
		if prompt == "" {
			return mcplib.NewToolResultError("prompt is required"), nil
		}

		rules, err := store.GetApplicableRules(ctx,
			req.GetString("team", ""),
			req.GetString("category", ""),
			req.GetString("user", ""),
		)
		if err != nil {
			return mcplib.NewToolResultError(fmt.Sprintf("get rules failed: %v", err)), nil
		}

		if len(rules) == 0 {
			return mcplib.NewToolResultText(prompt), nil
		}

		return mcplib.NewToolResultText(buildEnhancedPrompt(prompt, rules)), nil
	}
}

func buildEnhancedPrompt(prompt string, rules []storage.Rule) string {
	var sb strings.Builder
	sb.WriteString("The following rules apply to this request:\n\n")

	currentLabel := ""
	idx := 1
	for _, r := range rules {
		label := ruleScopeLabel(r)
		if label != currentLabel {
			if currentLabel != "" {
				sb.WriteString("\n")
			}
			fmt.Fprintf(&sb, "**%s:**\n", label)
			currentLabel = label
			idx = 1
		}
		fmt.Fprintf(&sb, "%d. %s\n", idx, r.Content)
		idx++
	}

	sb.WriteString("\n---\n\n")
	sb.WriteString(prompt)
	return sb.String()
}

func ruleScopeLabel(r storage.Rule) string {
	switch r.Scope {
	case storage.RuleScopeTeam:
		if r.ScopeValue != "" {
			return fmt.Sprintf("Team Rules (%s)", r.ScopeValue)
		}
		return "Team Rules"
	case storage.RuleScopeCategory:
		return fmt.Sprintf("Category Rules (%s)", r.ScopeValue)
	case storage.RuleScopeUser:
		if r.ScopeValue != "" {
			return fmt.Sprintf("User Rules (%s)", r.ScopeValue)
		}
		return "User Rules"
	default:
		return "Rules"
	}
}
```
